  const sidebarLinks = document.querySelectorAll('.sidebar-link');

    members: ['إدارة الأعضاء', 'عرض جميع الأعضاء وتسجيل أعضاء جدد'],

  };

  sidebarLinks.forEach(link => {
    link.addEventListener('click', () => {
      const key = link.getAttribute('data-section');
      // تفعيل الرابط
      sidebarLinks.forEach(l => l.classList.remove('active'));
      link.classList.add('active');
      // إظهار القسم
      Object.keys(sections).forEach(k => {
        sections[k].classList.toggle('active', k === key);
      });
      const [t, s] = titles[key];
      document.getElementById('page-title').textContent = t;
      document.getElementById('page-subtitle').textContent = s;
    });
  });

  // بيانات أولية للأعضاء
  let members = [
    {name:'أحمد علي', id:'#1023', start:'01 / 11 / 2025', end:'01 / 12 / 2025', status:'ساري', balance:'0 د.ب'},
    {name:'محمد سالم', id:'#1044', start:'05 / 10 / 2025', end:'05 / 11 / 2025', status:'منتهي', balance:'20 د.ب مستحق'},
    {name:'سعيد حسن', id:'#0999', start:'15 / 10 / 2025', end:'15 / 11 / 2025', status:'ينتهي قريباً', balance:'0 د.ب'}
  ];

  function renderMembers() {
    const tbody = document.getElementById('membersTable');
    const search = document.getElementById('searchMember').value.trim();
    tbody.innerHTML = '';
    members
      .filter(m => !search || m.name.includes(search) || m.id.includes(search))
      .forEach(m => {
        const tr = document.createElement('tr');
        const badgeClass =
          m.status === 'ساري' ? 'badge badge-active' :
          m.status === 'منتهي' ? 'badge badge-expired' :
          'badge badge-soon';
        tr.innerHTML = `
          <td>${m.name}</td>
          <td>${m.id}</td>
          <td>${m.start}</td>
          <td>${m.end}</td>
          <td><span class="${badgeClass}">${m.status}</span></td>
          <td>${m.balance}</td>
        `;
        tbody.appendChild(tr);
      });
  }

  document.getElementById('searchMember').addEventListener('input', renderMembers);

  function addMember() {
    const name = document.getElementById('newName').value.trim();
    const id = document.getElementById('newId').value.trim();
    const phone = document.getElementById('newPhone').value.trim();
    const age = document.getElementById('newAge').value.trim();
    const health = document.getElementById('newHealth').value.trim();

    if (!name || !id || !phone) {
      alert('الرجاء إدخال الاسم، رقم العضوية، ورقم الجوال على الأقل.');
      return;
    }

    members.push({
      name,
      id,
      start: '--',
      end: '--',
      status: 'ساري',
      balance: '0 د.ب'
    });

    document.getElementById('newName').value = '';
    document.getElementById('newId').value = '';
    document.getElementById('newPhone').value = '';
    document.getElementById('newAge').value = '';
    document.getElementById('newHealth').value = '';

    renderMembers();
    alert('تمت إضافة العضو للقائمة (عرض تجريبي فقط بدون حفظ في قاعدة بيانات).');
  }

  function addAttendance(type) {
    const id = document.getElementById('attMember').value.trim();
    const date = document.getElementById('attDate').value || today.toLocaleDateString('ar-EG', opts);
    if (!id) {
      alert('الرجاء إدخال رقم العضوية.');
      return;
    }
    const time = new Date().toLocaleTimeString('ar-EG', {hour:'2-digit', minute:'2-digit'});
    const tbody = document.getElementById('attTable');
    const tr = document.createElement('tr');
    tr.innerHTML = `<td>${time}</td><td>${date}</td><td>${id}</td><td>${type}</td>`;
    tbody.appendChild(tr);
    document.getElementById('attMember').value = '';
  }

  // Finance
  function parseNumber(value) {
    return parseFloat(String(value).replace(/,/g, '')) || 0;
  }

  function recalcNet() {
    const income = parseNumber(document.getElementById('incomeInput').value);
    const cost = parseNumber(document.getElementById('costInput').value);
    const net = income - cost;
    document.getElementById('netInput').value = net.toString();
  }
